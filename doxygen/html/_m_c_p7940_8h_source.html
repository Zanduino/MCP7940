237  /*************************************************************************************************************
238  ** Declare the readRAM() and writeRAM() methods as template functions to use for all I2C device I/O. The **
239  ** code has to be in the main library definition rather than the actual MCP7940.cpp library file. **
240  ** The template functions allow any type of data to be read and written, be it a byte or a character array **
241  ** or a structure. **
242  ** **
243  ** The MCP7940 supports 64 bytes of general purpose SRAM memory, which can be used to store data. For more **
244  ** details, see datasheet p.36. **
245  ** The data is stored in a block of 64 bytes, reading beyond the end of the block causes the address **
246  ** pointer to roll over to the start of the block. **
247  *************************************************************************************************************/
248  template< typename T > // method to read a structure //
249  uint8_t& readRAM(const uint8_t addr, T &value) { // //
250  uint8_t* bytePtr = (uint8_t*)&value; // Pointer to structure beginning //
251  uint8_t structSize = sizeof(T); // Number of bytes in structure //
252  uint8_t i = 0; // loop counter //
253  Wire.beginTransmission(MCP7940_ADDRESS); // Address the I2C device //
254  Wire.write((addr%64) + MCP7940_RAM_ADDRESS); // Send register address to write //
255  _TransmissionStatus = Wire.endTransmission(); // Close transmission //
256  for (i=0;i<structSize;i++) { // loop for each byte to be read //
257  if (i%BUFFER_LENGTH==0) { // Read I2C again on buffer boundary//
258  Wire.requestFrom(MCP7940_ADDRESS, structSize); // Request a block of data //
259  } // of if-then we are at a buffer boundary // //
260  *bytePtr++ = Wire.read(); // //
261  } // of for-next each byte to be read // //
262  return (i); // return bytes read //
263  } // of method readRAM() //----------------------------------//
264  template<typename T> // method to write any data type to //
265  bool writeRAM(const uint8_t addr, const T &value) { // the MCP7940 SRAM //
266  const uint8_t* bytePtr = (const uint8_t*)&value; // Pointer to structure beginning //
267  Wire.beginTransmission(MCP7940_ADDRESS); // Address the I2C device //
268  Wire.write((addr%64) + MCP7940_RAM_ADDRESS); // Send register address to write //
269  for (uint8_t i = 0; i < sizeof(T); i++) { // loop for each byte to be written //
270  Wire.write(*bytePtr++); // //
271  } // //
272  _TransmissionStatus = Wire.endTransmission(); // Close transmission //
273  return (!_TransmissionStatus); // return error status //
274  } // of method writeRAM() //----------------------------------//
